# Переменные среды для запуска
```
# В cura используется postgres, поэтому url базы данных собирается внутри кода
DB_USER=
DB_PASSWORD=
DB_NAME=

# Пароль для шифрования куки сессии длинной 32 символа
NUXT_SESSION_PASSWORD=
```
